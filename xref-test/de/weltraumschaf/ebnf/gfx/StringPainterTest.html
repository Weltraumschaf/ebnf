33      @Test public void calcXPosition() {

36          assertEquals(5, painter.calcXPosition(10, -3, 20));

38          assertEquals(33, painter.calcXPosition(233, -3, 300));

40          assertEquals(0, painter.calcXPosition(10, 0, 10));
41          assertEquals(5, painter.calcXPosition(10, 0, 20));

45          assertEquals(2, painter.calcXPosition(10, 2, 10));

59      @Test public void drawCenteredString() {
60          final String str = "test";
61          final int stringWidth = 20;
62          final int width = 100;
63          final int height = 100;
64          final int ascent = 5;
65          final int descent = 10;
66          final Font font = Fonts.SANSERIFIT.create();
67          final Font backup = Fonts.SANSERIF.create();
68          final FontMetrics metrics = mock(FontMetrics.class);
69          when(metrics.stringWidth(str)).thenReturn(stringWidth);
70          when(metrics.getAscent()).thenReturn(ascent);
71          when(metrics.getDescent()).thenReturn(descent);
72  
73          final Graphics2D graphics = mock(Graphics2D.class);
74          when(graphics.getFont()).thenReturn(backup);
75          when(graphics.getFontMetrics()).thenReturn(metrics);
76  
77          final StringPainter painter = new StringPainter(graphics, font);
78          painter.drawCenteredString(str, new Dimension(width, 100));
79          verify(graphics).getFont();
80          verify(graphics).setFont(font);
81          verify(metrics).stringWidth(str);
82          verify(metrics).getAscent();
83          verify(metrics).getDescent();
84          verify(graphics).drawString(str,
85                                      painter.calcXPosition(stringWidth, 0, width),
86                                      painter.calcYPosition(0, height, ascent, descent));
87      }
