25  public class LoopTest {

27      private void assertInitialGrid(final Loop loop) {
28          assertBaseGrid(loop);
29          assertTrue(loop.grid.get(1, 0) instanceof Empty);
30          assertTrue(loop.grid.get(1, 1) instanceof StraightWE);
31      }
32  
33      private void assertBaseGrid(final Loop loop) {
34          assertEquals(2, loop.grid.counRows());
35          assertEquals(3, loop.grid.countCols());
36          assertTrue(loop.grid.get(0, 0) instanceof ColumnLayout);
37          // col 2 differs
38          assertTrue(loop.grid.get(2, 0) instanceof ColumnLayout);
39          // 2. row
40          assertTrue(loop.grid.get(0, 1) instanceof ColumnLayout);
41          // col 2 differs if additional set
42          assertTrue(loop.grid.get(2, 1) instanceof ColumnLayout);
43  
44          assertSplitAndJoin(loop);
45          assertCurves(loop);
46      }
47  
48      private void assertSplitAndJoin(final Loop loop) {
49          final ColumnLayout split = (ColumnLayout)loop.grid.get(0, 0);
50          assertEquals(1, split.countShapes());
51          assertTrue(split.get(0) instanceof HForkSE);
52  
53          final ColumnLayout join  = (ColumnLayout)loop.grid.get(2, 0);
54          assertEquals(1, join.countShapes());
55          assertTrue(join.get(0) instanceof HForkSW);
56      }
57  
58      private void assertCurves(final Loop loop) {
59          final ColumnLayout firstCurve = (ColumnLayout)loop.grid.get(0, 1);
60          assertEquals(1, firstCurve.countShapes());
61          assertTrue(firstCurve.get(0) instanceof CurveNE);
62  
63          final ColumnLayout lastCurve  = (ColumnLayout)loop.grid.get(2, 1);
64          assertEquals(1, lastCurve.countShapes());
65          assertTrue(lastCurve.get(0) instanceof CurveNW);
66      }
67  
68      @Test public void setLooped() {
69          final Loop loop = loop();
70          assertInitialGrid(loop);
71  
72          loop.setLooped(terminal("foo"));
73          assertBaseGrid(loop);
74          assertTrue(loop.grid.get(1, 0) instanceof Terminal);
75          assertTrue(loop.grid.get(1, 1) instanceof StraightWE);
76  
77          final Shape greatOne = terminal("bar");
78          greatOne.setSize(new Dimension(Shape.DEFAULT_WIDTH, Shape.DEFAULT_HEIGHT * 3));
79          loop.setLooped(greatOne);
80          assertTrue(loop.grid.get(1, 0) instanceof Terminal);
81          assertTrue(loop.grid.get(1, 1) instanceof StraightWE);
82  
83          final ColumnLayout split = (ColumnLayout)loop.grid.get(0, 0);
84          assertEquals(3, split.countShapes());
85          assertTrue(split.get(0) instanceof HForkSE);
86          assertTrue(split.get(1) instanceof StraightNS);
87          assertTrue(split.get(2) instanceof StraightNS);
88          final ColumnLayout join  = (ColumnLayout)loop.grid.get(2, 0);
89          assertEquals(3, join.countShapes());
90          assertTrue(join.get(0) instanceof HForkSW);
91          assertTrue(join.get(1) instanceof StraightNS);
92          assertTrue(join.get(2) instanceof StraightNS);
93          assertCurves(loop);
94      }
