54      @Test public void backupAndRestoreColorAndStroke() {
55          final Color backupedColor = Color.RED;
56          final Stroke backupedStroke = mock(Stroke.class);
57          final Graphics2D graphics = mock(Graphics2D.class);
58          when(graphics.getColor()).thenReturn(backupedColor);
59          when(graphics.getStroke()).thenReturn(backupedStroke);
60          final AbstractShape sut = new AbstractShapeImpl();
61          assertEquals(0, sut.countBackup());
62          sut.backupColorAndStroke(graphics);
63          assertEquals(1, sut.countBackup());
64          final GraphicsSetting backup = sut.getBackup().peek();
65          assertEquals(backupedColor, backup.getColor());
66          assertSame(backupedStroke, backup.getStroke());
67          sut.resotreColorAndStroke(graphics);
68          sut.resotreColorAndStroke(graphics);
69          assertEquals(0, sut.countBackup());
70          verify(graphics, times(1)).setColor(backup.getColor());
71          verify(graphics, times(1)).setStroke(backup.getStroke());
72      }

74      @Test public void debugShape() {
75          final Graphics2D notInvokedGfx = mock(Graphics2D.class);
76          final Paintable notInvokedPaintable = mock(Paintable.class); // NOPMD
77          final AbstractShape noDebug = new AbstractShapeImpl();
78          assertFalse(noDebug.isDebug());
79          noDebug.debugShape(notInvokedGfx, notInvokedPaintable);
80          verify(notInvokedGfx, never()).setColor(Color.RED);
81          verify(notInvokedGfx, never()).setStroke(Strokes.createForDebug());
82          verify(notInvokedPaintable, never()).paint(notInvokedGfx);
83  
84          final Graphics2D invokedGfx = mock(Graphics2D.class);
85          final Paintable invokedPaintable = mock(Paintable.class);
86          final AbstractShape debug = new AbstractShapeImpl();
87          debug.setDebug(true);
88          assertTrue(debug.isDebug());
89          debug.debugShape(invokedGfx, invokedPaintable);
90          verify(invokedGfx, times(1)).setColor(Color.RED);
91          verify(invokedGfx, times(1)).setStroke(Strokes.createForDebug());
92          verify(invokedPaintable, times(1)).paint(invokedGfx);
