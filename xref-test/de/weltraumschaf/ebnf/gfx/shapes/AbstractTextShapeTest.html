58      @Test public void calculateWidth() {
59          try {
60              AbstractTextShape.calculateWidth(-1);
61              fail("Expected exception not thrown!");
62          } catch (IllegalArgumentException ex) {
63              assertEquals("box width need to be greater or equal zero!", ex.getMessage());
64          }
65  
66          assertEquals(1 * Shape.DEFAULT_WIDTH, AbstractTextShape.calculateWidth(0));
67          assertEquals(1 * Shape.DEFAULT_WIDTH, AbstractTextShape.calculateWidth(1));
68          assertEquals(1 * Shape.DEFAULT_WIDTH, AbstractTextShape.calculateWidth(15));
69          assertEquals(1 * Shape.DEFAULT_WIDTH,
70                       AbstractTextShape.calculateWidth(Shape.DEFAULT_WIDTH - 11));
71          assertEquals(2 * Shape.DEFAULT_WIDTH,
72                       AbstractTextShape.calculateWidth(Shape.DEFAULT_WIDTH));
73          assertEquals(3 * Shape.DEFAULT_WIDTH,
74                       AbstractTextShape.calculateWidth(3 * Shape.DEFAULT_WIDTH - 11));
75          assertEquals(4 * Shape.DEFAULT_WIDTH,
76                       AbstractTextShape.calculateWidth(3 * Shape.DEFAULT_WIDTH));
77      }

79      @Test public void createStringPainter() {
80          final Graphics2D graphics1 = mock(Graphics2D.class);
81          final Graphics2D graphics2 = mock(Graphics2D.class);
82          final AbstractTextShape sut = new AbstractTextShapeStub("");
83          StringPainter painter = sut.createStringPainter(graphics1);
84          assertSame(painter, sut.createStringPainter(graphics1));
85          assertSame(painter, sut.createStringPainter(graphics1));
86          assertNotSame(painter, sut.createStringPainter(graphics2));
87          painter = sut.createStringPainter(graphics2);
88          assertSame(painter, sut.createStringPainter(graphics2));
89          assertNotSame(painter, sut.createStringPainter(graphics1));
90      }

92      @Test public void calculateTextSize() {

94          final FontRenderContext context = mock(FontRenderContext.class);

96          final Font font = mock(Font.class);

99          when(font.getStringBounds(str, context)).thenReturn(rect);
100         final AbstractTextShape sut = new AbstractTextShapeStub(str, font);
101         assertEquals(new Dimension(10, 20), sut.calculateTextSize(graphics));
