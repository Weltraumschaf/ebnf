33      @Test public void testCreateRow() {

47      @Test public void testGenerateMatrix() {
48          Syntax ast = syntax("fobar")
49              .rule("one")
50              .end()
51              .rule("two")
52              .end()
53              .rule("three")
54              .end()
55          .build();
56  
57          assertEquals(2, ast.depth());
58          final TextSyntaxTree visitor = new TextSyntaxTree();
59          assertEquals(new ArrayList<ArrayList<String>>(), visitor.getMatrix());
60          assertEquals(0, visitor.getDepth());
61          ast.accept(visitor);
62  
63          assertEquals(2, visitor.getDepth());
64          final List<List<String>> matrix = visitor.getMatrix();
65          assertEquals(4, matrix.size());
66          assertEquals("[syntax]",  matrix.get(0).get(0));
67          assertEquals("",          matrix.get(0).get(1));
68          assertEquals(" +--",       matrix.get(1).get(0)); // NOPMD
69          assertEquals("[rule='one']", matrix.get(1).get(1));
70          assertEquals(" +--",       matrix.get(2).get(0));
71          assertEquals("[rule='two']", matrix.get(2).get(1));
72          assertEquals(" +--",       matrix.get(3).get(0));
73          assertEquals("[rule='three']", matrix.get(3).get(1));
74  
75          ast = syntax("foo") // NOPMD
76              .rule("literal")
77                  .choice()
78                      .sequence()
79                          .terminal("'")
80                          .identifier("character") // NOPMD
81                          .loop()
82                              .identifier("character")
83                          .end()
84                          .terminal("'")
85                      .end()
86                      .sequence()
87                          .terminal("\"")
88                          .identifier("character")
89                          .loop()
90                              .identifier("character")
91                          .end()
92                          .terminal("\"")
93                      .end()
94                  .end()
95              .end()
96              .build();
97  
98          ast.accept(visitor);
99          assertEquals(6, visitor.getDepth());
100         @SuppressWarnings("unchecked")
101         final List<List<String>> expected = Arrays.asList(
102             Arrays.asList("[syntax]", "", "", "", "", ""),
103             Arrays.asList(" +--",     "[rule='literal']", "", "", "", ""),
104             Arrays.asList("    ",     " +--", "[choice]", "", "", ""),
105             Arrays.asList("    ",     "    ", " +--", "[sequence]", "", ""),
106             Arrays.asList("    ",     "    ", " |  ", " +--",       "[terminal=''']", ""),
107             Arrays.asList("    ",     "    ", " |  ", " +--",       "[identifier='character']", ""),
108             Arrays.asList("    ",     "    ", " |  ", " +--",       "[loop]", ""),
109             Arrays.asList("    ",     "    ", " |  ", " |  ",       " +--", "[identifier='character']"),
110             Arrays.asList("    ",     "    ", " |  ", " +--",       "[terminal=''']", ""),
111             Arrays.asList("    ",     "    ", " +--", "[sequence]", "", ""),
112             Arrays.asList("    ",     "    ", "    ", " +--",       "[terminal='\"']", ""),
113             Arrays.asList("    ",     "    ", "    ", " +--",       "[identifier='character']", ""),
114             Arrays.asList("    ",     "    ", "    ", " +--",       "[loop]", ""),
115             Arrays.asList("    ",     "    ", "    ", " |  ",       " +--", "[identifier='character']"),
116             Arrays.asList("    ",     "    ", "    ", " +--",       "[terminal='\"']", "")
117         );
118         assertEquals(expected, visitor.getMatrix());
119     }

121     @Test public void testGenerateText() throws URISyntaxException, IOException, SyntaxException {
122         TextSyntaxTree visitor = new TextSyntaxTree();
123         Syntax ast = syntax("foo").build();
124         ast.accept(visitor);
125         assertEquals("[syntax]\n", visitor.getText()); // NOPMD
126 
127         ast = syntax("foo")
128             .rule("rule-1")
129             .end()
130         .build();
131         ast.accept(visitor);
132         assertEquals("[syntax]\n"
133                    + " +--[rule='rule-1']\n",
134             visitor.getText()
135         );
136 
137         ast = syntax("foo")
138             .rule("rule-1")
139             .end()
140             .rule("rule-2")
141             .end()
142         .build();
143         ast.accept(visitor);
144         assertEquals("[syntax]\n"
145                    + " +--[rule='rule-1']\n"
146                    + " +--[rule='rule-2']\n",
147             visitor.getText()
148         );
149 
150         ast = syntax("foo")
151             .rule("name")
152                 .choice()
153                 .end()
154             .end()
155         .build();
156         ast.accept(visitor);
157         assertEquals("[syntax]\n"
158                    + " +--[rule='name']\n"
159                    + "     +--[choice]\n",
160             visitor.getText()
161         );
162 
163         ast = syntax("foo")
164             .rule("name")
165                 .choice()
166                     .identifier("ident")
167                     .terminal("term")
168                 .end()
169             .end()
170         .build();
171         ast.accept(visitor);
172         assertEquals("[syntax]\n"
173                    + " +--[rule='name']\n"
174                    + "     +--[choice]\n"
175                    + "         +--[identifier='ident']\n"
176                    + "         +--[terminal='term']\n",
177             visitor.getText()
178         );
179 
180         ast = syntax("foobar") // NOPMD
181             .rule("one")
182                 .choice()
183                 .end()
184             .end()
185             .rule("two")
186             .end()
187         .build();
188         ast.accept(visitor);
189         assertEquals("[syntax]\n"
190                    + " +--[rule='one']\n"
191                    + " |   +--[choice]\n"
192                    + " +--[rule='two']\n",
193             visitor.getText()
194         );
195 
196         final String expected = helper().createStringFromFixture("visitor/rules_with_literals_tree_output");
197         final Parser parser = helper().createParserFromFixture("rules_with_literals.ebnf");
198         ast     = parser.parse();
199         visitor = new TextSyntaxTree();
200         ast.accept(visitor);
201         assertEquals(
202             expected,
203             visitor.getText()
204         );
205     }
