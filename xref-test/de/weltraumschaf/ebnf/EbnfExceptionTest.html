14  import java.io.IOException;
15  import static org.junit.Assert.*;
16  import org.junit.Test;

18  /**

20   * @author Sven Strittmatter <weltraumschaf@googlemail.com>

22  public class EbnfExceptionTest {
23  
24      @Test public void creation() {
25          EbnfException sut = new EbnfException("foo");
26          assertEquals("foo", sut.getMessage());
27          assertEquals(App.ExitCode.FATAL_ERROR.getCode(), sut.getCode());
28          assertNull(sut.getCause());
29  
30          sut = new EbnfException("bar", App.ExitCode.NO_SYNTAX);
31          assertEquals("bar", sut.getMessage());
32          assertEquals(App.ExitCode.NO_SYNTAX.getCode(), sut.getCode());
33          assertNull(sut.getCause());
34  
35          sut = new EbnfException("baz", 5);
36          assertEquals("baz", sut.getMessage());
37          assertEquals(5, sut.getCode());
38          assertNull(sut.getCause());
39  
40          final Throwable cause = new IOException();
41          sut = new EbnfException("snafu", 5, cause);
42          assertEquals("snafu", sut.getMessage());
43          assertEquals(5, sut.getCode());
44          assertSame(cause, sut.getCause());
45  
46          sut = new EbnfException("bla", App.ExitCode.NO_SYNTAX, cause);
47          assertEquals("bla", sut.getMessage());
48          assertEquals(App.ExitCode.NO_SYNTAX.getCode(), sut.getCode());
49          assertSame(cause, sut.getCause());
50      }
