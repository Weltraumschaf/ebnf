28  public final class TestHelper {

43      public URI createResourceFromFixture(final String fixtureFile) throws URISyntaxException {
44          return getClass().getResource(FIXTURE_DIR + '/' + fixtureFile).toURI();
45      }
46  
47      public String createStringFromFixture(final String fixtureFile) throws URISyntaxException, IOException {
48          return Files.toString(new File(createResourceFromFixture(fixtureFile)), Charset.defaultCharset());
49      }
50  
51      public BufferedReader createSourceFromFixture(final String fixtureFile) throws FileNotFoundException,
52                                                                                URISyntaxException {
53          return ReaderHelper.createFrom(createResourceFromFixture(fixtureFile));
54      }
55  
56      public Scanner createScannerFromFixture(final String fixtureFile) throws FileNotFoundException, URISyntaxException {
57          return new Scanner(createSourceFromFixture(fixtureFile));
58      }
