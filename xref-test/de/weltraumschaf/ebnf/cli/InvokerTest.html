35      @Test public void parseOptions() throws EbnfException, ParseException {
36          final CliOptions options = mock(CliOptions.class);
37          final String[] args = new String[]{"-d", "-h"};
38          final Invoker sut = new Invoker(args, mock(PrintStream.class), options);
39          final CliOptions parsedOptions = sut.parseOptions();
40          assertSame(options, parsedOptions);
41          verify(options, times(1)).parse(args);
42      }

44      @Test public void parseOptionsWithPArseError() throws ParseException {
45          final CliOptions options = mock(CliOptions.class);
46          final String[] args = new String[]{"foo"};
47          final Invoker sut = new Invoker(args, mock(PrintStream.class), options);
48          final ParseException throwed = new ParseException("foobar");
49          doThrow(throwed).when(options).parse(args);
50  
51          try {
52              sut.parseOptions();
53              fail("Expected exeption not thrown!");
54          } catch (EbnfException ex) {
55              assertEquals(throwed.getMessage(), ex.getMessage());
56              assertSame(throwed, ex.getCause());
57          }

59          verify(options, times(1)).parse(args);
60      }
