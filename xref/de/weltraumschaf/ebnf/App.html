92      private ExitCode run()  throws EbnfException {
93          final CliOptions options = parseOptions();
94  
95          if (options.isHelp()) {
96              options.format(new HelpFormatter());
97              return ExitCode.OK;
98          }
99  
100         if (!options.hasSyntaxFile()) {
101             System.out.println("No syntax file given!");
102             return ExitCode.NO_SYNTAX;
103         }
104 
105         Scanner scanner;
106         Parser parser;
107         Syntax ast;
108 
109         try {
110             scanner = new Scanner(ReaderHelper.createFrom(new File(options.getSyntaxFile())));
111             parser  = new Parser(scanner);
112             ast     = parser.parse();
113         } catch (FileNotFoundException ex) {
114             System.out.println(String.format("Can not read syntax file '%s'!", options.getSyntaxFile()));
115 
116             if (options.isDebug()) {
117                 ex.printStackTrace(System.out);
118             }
119 
120             return ExitCode.READ_ERROR;
121         } catch (IOException ex) {
122             System.out.println(String.format("Can not read syntax file '%s'!", options.getSyntaxFile()));
123 
124             if (options.isDebug()) {
125                 ex.printStackTrace(System.out);
126             }
127 
128             return ExitCode.READ_ERROR;
129         }
130 
131         if (options.isTextTree()) {
132             final TextSyntaxTree visitor = new TextSyntaxTree();
133             ast.accept(visitor);
134             System.out.println(visitor.getText());
135         }
136 
137         return ExitCode.OK;
138     }
