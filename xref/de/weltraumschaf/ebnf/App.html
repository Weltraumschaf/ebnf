47      private ExitCode run()  throws EbnfException {
48          final CliOptions options = parseOptions();
49  
50          if (options.isHelp()) {
51              options.format(new HelpFormatter());
52              return ExitCode.OK;
53          }
54  
55          if (!options.hasSyntaxFile()) {
56              println("No syntax file given!");
57              return ExitCode.NO_SYNTAX;
58          }
59  
60          Scanner scanner;
61          Parser parser;
62          Syntax ast;
63  
64          try {
65              scanner = new Scanner(ReaderHelper.createFrom(new File(options.getSyntaxFile())));
66              parser  = new Parser(scanner);
67              ast     = parser.parse();
68          } catch (FileNotFoundException ex) {
69              println(String.format("Can not read syntax file '%s'!", options.getSyntaxFile()));
70  
71              if (options.isDebug()) {
72                  ex.printStackTrace(System.out);
73              }
74  
75              return ExitCode.READ_ERROR;
76          } catch (IOException ex) {
77              println(String.format("Can not read syntax file '%s'!", options.getSyntaxFile()));
78  
79              if (options.isDebug()) {
80                  ex.printStackTrace(System.out);
81              }
82  
83              return ExitCode.READ_ERROR;
84          }
85  
86          if (options.isTextTree()) {
87              final TextSyntaxTree visitor = new TextSyntaxTree();
88              ast.accept(visitor);
89              println(visitor.getText());
90          }
91  
92          return ExitCode.OK;
93      }
94  
95      private CliOptions parseOptions() throws EbnfException {
96          final CliOptions options = new CliOptions();
97  
98          try {
99              options.parse(args);
100         } catch (ParseException ex) {
101             throw new EbnfException(ex.getMessage(), 1, ex);
102         }
103 
104         return options;
105     }
