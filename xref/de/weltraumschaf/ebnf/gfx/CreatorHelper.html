28      public GridLayout createValueDiagram(final Graphics2D graphics) {
29          final GridLayout value = grid();
30          value.append(
31              column().append(
32                  rule("value"),
33                  start(),
34                  empty(),
35                  empty(),
36                  empty(),
37                  empty()
38              )
39          );
40          value.set(1, 1, choice(
41              identifier("string"),
42              identifier("number"),
43              identifier("object"),
44              terminal("true"),
45              terminal("false")
46          ));
47          value.append(
48              column().append(
49                  empty(),
50                  end(),
51                  empty(),
52                  empty(),
53                  empty(),
54                  empty()
55              )
56          );
57          value.adjust(graphics);
58          return value;
59      }
60  
61      public GridLayout createObjectDiagram(final Graphics2D graphics) {
62          final GridLayout object = grid();
63          final Sequence innerSequence = sequence(identifier("string"),
64                                                  terminal(":"),
65                                                  identifier("value"));
66          innerSequence.adjust(graphics);
67          final Loop loop = loop(innerSequence, terminal(","));
68          loop.adjust(graphics);
69          final Option option = option(loop);
70  
71          object.append(
72              column().append(
73                  rule("object"),
74                  start(),
75                  empty(),
76                  empty()
77              )
78          );
79          object.append(
80              column().append(
81                  empty(),
82                  terminal("{"),
83                  empty(),
84                  empty()
85              )
86          );
87          object.append(
88              column().append(
89                  empty(),
90                  option
91              )
92          );
93          object.append(
94              column().append(
95                  empty(),
96                  terminal("}"),
97                  empty(),
98                  empty()
99              )
100         );
101         object.append(
102             column().append(
103                 empty(),
104                 end(),
105                 empty(),
106                 empty()
107             )
108         );
109         object.adjust(graphics);
110         return object;
111     }
112 
113     public RailroadDiagram createDiagram(final Graphics graphics) {
114         return createDiagram((Graphics2D) graphics);
115     }
116 
117     public RailroadDiagram createDiagram(final Graphics2D graphics) {
118         final Point offset = new Point(20, 20);
119         final RailroadDiagram diagram = new RailroadDiagram();
120         final GridLayout value = createValueDiagram(graphics);
121         value.setPosition(offset);
122         diagram.add(value);
123 
124         final GridLayout object = createObjectDiagram(graphics);
125         object.setPosition(new Point(offset.x, offset.y + value.getSize().height));
126         diagram.add(object);
127 
128         return diagram;
129     }
