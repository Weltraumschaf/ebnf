11  package de.weltraumschaf.ebnf.gfx;
12  
13  import java.awt.Dimension;
14  import java.awt.Font;
15  import java.awt.FontMetrics;
16  import java.awt.Graphics2D;
17  
18  /**
19   * Paints a centered string into a rectangular area.
20   *
21   * Also provides some helpers for measuring text and calculate positions.
22   *
23   * The algorithm for position the text as centered is from:
24   * http://stackoverflow.com/questions/5378052/positioning-string-in-graphic-java
25   *
26   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
27   */
28  public class StringPainter {
29  
30      /**
31       * Default font for terminal shapes.
32       */
33      public static final Font MONOSPACED = Fonts.MONOSPACED.create();
34      /**
35       * Default font for rule shapes.
36       */
37      public static final Font SANSERIF   = Fonts.SANSERIF.create();
38      /**
39       * Default font for identifier shapes.
40       */
41      public static final Font SANSERIFIT = Fonts.SANSERIFIT.create();
42  
43      /**
44       * Default graphics object.
45       */
46      private static final Graphics2D DEFAULT_GRAPHIC = Helper.newGraphics();
47  
48      /**
49       * used graphics object.
50       */
51      private final Graphics2D graphic;
52      /**
53       * Used font object.
54       */
55      private final Font font;
56  
57      public StringPainter() {
58          this(DEFAULT_GRAPHIC);
59      }
60  
61      public StringPainter(final Font font) {
62          this(DEFAULT_GRAPHIC, font);
63      }
64  
65      public StringPainter(final Graphics2D graphic) {
66          this(graphic, MONOSPACED);
67      }
68  
69      public StringPainter(final Graphics2D graphic, final Font font) {
70          this.graphic  = graphic;
71          this.font     = font;
72      }
73  
74      /**
75       * Draws a string centered inside the given size.
76       *
77       * @param str  String to center.
78       * @param size Size of the rectangle to center inside.
79       */
80      public void drawCenteredString(final String str, final Dimension size) {
81          drawCenteredString(str, new Point(), size);
82      }
83  
84      /**
85       * Draws a string centered inside the given size with an x and y offset..
86       *
87       * @param str    String to center.
88       * @param offset X and y offset.
89       * @param size   Size of the rectangle to center inside.
90       */
91      public void drawCenteredString(final String str, final Point offset, final Dimension size) {
92          final Font backup = graphic.getFont();
93          graphic.setFont(font);
94          final FontMetrics metrics = graphic.getFontMetrics();
95          final int xPosition = calcXPosition(metrics.stringWidth(str), offset.x, size.width);
96          final int yPosition = calcYPosition(offset.y, size.height, metrics.getAscent(), metrics.getDescent());
97          graphic.drawString(str, xPosition, yPosition);
98          graphic.setFont(backup);
99      }
100 
101     /**
102      * Calculates the string x position to draw.
103      *
104      * @param stringWidth
105      * @param offsetX
106      * @param width
107      * @return
108      */
109     public int calcXPosition(final int stringWidth, final int offsetX, final int width) {
110         final int _offsetX = offsetX < 0
111                            ? 0
112                            : offsetX;
113         return (width - stringWidth) / 2 + _offsetX;
114     }
115 
116     /**
117      * Calculates the string y position to draw.
118      * 
119      * @param offsetY
120      * @param height
121      * @param ascent
122      * @param descent
123      * @return
124      */
125     public int calcYPosition(final int offsetY, final int height, final int ascent, final int descent) {
126         return (ascent + (height - (ascent + descent)) / 2) + offsetY;
127     }
128 }
