12  package de.weltraumschaf.ebnf.gfx.shapes;
13  
14  import com.google.common.collect.Lists;
15  import de.weltraumschaf.ebnf.gfx.Point;
16  import static de.weltraumschaf.ebnf.gfx.shapes.ShapeFactory.empty;
17  import java.awt.Dimension;
18  import java.awt.Graphics2D;
19  import java.util.List;
20  
21  /**
22   *
23   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
24   */
25  public class Sequence extends AbstractLayout {
26  
27      private final List<Shape> row = Lists.newArrayList();
28  
29      public Shape get(final int index) {
30          return row.get(index);
31      }
32  
33      public Sequence set(final int index, final Shape shape) {
34          final int count = countShapes();
35  
36          if (index >= count) {
37              append(empty(count - 1));
38              append(shape);
39          } else {
40              row.set(index, shape);
41          }
42  
43          return this;
44      }
45  
46      public Sequence append(final Shape... shapes) {
47          for (Shape shape : shapes) {
48              append(shape);
49          }
50  
51          return this;
52      }
53  
54      public Sequence append(final Shape shape) {
55          row.add(shape);
56          return this;
57      }
58  
59      @Override
60      public void paint(final Graphics2D graphic) {
61          if (countShapes() == 0) {
62              return;
63          }
64  
65          final Point pos = getPosition();
66          int currentX = pos.x;
67  
68          for (Shape shape : row) {
69              shape.setPosition(pos.setX(currentX));
70              shape.setDebug(isDebug());
71              shape.paint(graphic);
72              currentX += shape.getSize().width;
73          }
74      }
75  
76      public int countShapes() {
77          return row.size();
78      }
79  
80      @Override
81      public void adjust(final Graphics2D graphics) {
82          int width = 0;
83          int height = 0;
84  
85          for (Shape shape : row) {
86              adjustShape(shape, graphics);
87              final Dimension shapeSize = shape.getSize();
88              width += shapeSize.width;
89              height = Math.max(height, shapeSize.height);
90          }
91  
92          for (Shape shape : row) {
93              shape.getSize().height = height;
94          }
95  
96          setSize(new Dimension(width, height));
97      }
98  
99  }
