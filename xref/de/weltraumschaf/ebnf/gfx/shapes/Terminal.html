12  package de.weltraumschaf.ebnf.gfx.shapes;
13  
14  import de.weltraumschaf.ebnf.gfx.Point;
15  import de.weltraumschaf.ebnf.gfx.StringPainter;
16  import de.weltraumschaf.ebnf.gfx.Strokes;
17  import java.awt.Color;
18  import java.awt.Dimension;
19  import java.awt.Graphics2D;
20  
21  /**
22   *
23   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
24   */
25  public class Terminal extends AbstractTextShape implements Shape {
26  
27      private static final int ARC_WIDTH = 25;
28  
29      public Terminal(final String text) {
30          super(text, StringPainter.MONOSPACED);
31      }
32  
33      private Dimension boxSize = null;
34  
35      protected Dimension calcBoxSize(final Graphics2D graphic) {
36          final Dimension textSize = calculateTextSize(graphic);
37          return new Dimension(textSize.width + H_PADDING * 4, textSize.height + 4);
38      }
39  
40      @Override
41      public void adjust(final Graphics2D graphic) {
42          boxSize = calcBoxSize(graphic);
43          setSize(new Dimension(calculateWidth(boxSize.width), DEFAULT_HEIGHT));
44      }
45  
46      @Override
47      public void paint(final Graphics2D graphic) {
48          if (null == boxSize) {
49              adjust(graphic);
50          }
51  
52          final Point pos     = getPosition();
53          final Dimension size = getSize();
54          final int vCenter   = getCenterY();
55          final int hPaddingp = (size.width - boxSize.width) / 2;
56          final int vPadding  = (size.height - boxSize.height) / 2;
57          final Point rectanglePosition = new Point(pos.x + hPaddingp, pos.y + vPadding),
58                      inLineStart       = new Point(pos.x, vCenter),
59                      inLineEnd         = new Point(pos.x + hPaddingp, vCenter),
60                      outLineStart      = new Point(pos.x + hPaddingp + boxSize.width, vCenter),
61                      outLineEnd        = new Point(pos.x + size.width, vCenter);
62  
63          super.paint(graphic);
64          backupColorAndStroke(graphic);
65          graphic.setColor(Color.BLACK);
66          graphic.setStroke(Strokes.createForBox());
67          graphic.drawRoundRect(rectanglePosition.x, rectanglePosition.y, boxSize.width, boxSize.height, ARC_WIDTH, ARC_WIDTH);
68          graphic.setStroke(Strokes.createForRail());
69          graphic.drawLine(inLineStart.x, inLineStart.y, inLineEnd.x, inLineEnd.y);
70          graphic.drawLine(outLineStart.x, outLineStart.y, outLineEnd.x, outLineEnd.y);
71          final StringPainter textPainter = createStringPainter(graphic);
72          textPainter.drawCenteredString(getText(), rectanglePosition, boxSize);
73          resotreColorAndStroke(graphic);
74      }
75  
76  }
