12  package de.weltraumschaf.ebnf.gfx.shapes;

14  import de.weltraumschaf.ebnf.gfx.Point;
15  import de.weltraumschaf.ebnf.gfx.ShapeFactory;
16  import de.weltraumschaf.ebnf.gfx.Strokes;
17  import java.awt.Color;
18  import java.awt.Dimension;
19  import java.awt.Graphics2D;
20  import java.awt.Stroke;
21  import java.util.Stack;
22  
23  /**
24   *
25   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
26   */
27  abstract public class AbstractShape implements Shape {
28  
29      protected static class GraphicsSetting {
30          private final Color color;
31          private final Stroke stroke;
32  
33          public GraphicsSetting(final Color color, final Stroke stroke) {
34              this.color  = color;
35              this.stroke = stroke;
36          }
37  
38          public Color getColor() {
39              return color;
40          }
41  
42          public Stroke getStroke() {
43              return stroke;
44          }
45      }
46  
47      private final Stack<GraphicsSetting> backup = new Stack<GraphicsSetting>();
48      private Point position;
49      private Dimension size;
50      private boolean debug;
51  
52      public AbstractShape() {
53          size     = new Dimension(DEFAULT_WIDTH, DEFAULT_HEIGHT);
54          position = new Point();
55      }
56  
57      @Override
58      public Point getPosition() {
59          return position;
60      }
61  
62      @Override
63      public void setPosition(final Point position) {
64          this.position = position;
65      }
66  
67      @Override
68      public void setDebug(final boolean onOff) {
69          debug = onOff;
70      }
71  
72      @Override
73      public Dimension getSize() {
74          return size;
75      }
76  
77      @Override
78      public void setSize(final Dimension size) {
79          this.size = size;
80      }
81  
82      protected boolean isDebug() {
83          return debug;
84      }
85  
86      protected int countBackup() {
87          return backup.size();
88      }
89  
90      protected Stack<GraphicsSetting> getBackup() {
91          return backup;
92      }
93  
94      protected void backupColorAndStroke(final Graphics2D graphic) {
95          backup.push(new GraphicsSetting(graphic.getColor(), graphic.getStroke()));
96      }
97  
98      protected void resotreColorAndStroke(final Graphics2D graphic) {
99          if (!backup.isEmpty()) {
100             final GraphicsSetting settings = backup.pop();
101             graphic.setColor(settings.getColor());
102             graphic.setStroke(settings.getStroke());
103         }
104     }
105 
106     protected void debugShape(final Graphics2D graphic, final Paintable paintable) {
107         if (isDebug()) {
108             backupColorAndStroke(graphic);
109             graphic.setColor(Color.RED);
110             graphic.setStroke(Strokes.createForDebug());
111             paintable.paint(graphic);
112             resotreColorAndStroke(graphic);
113         }
114     }
115 
116     protected int getCenterX() {
117         return getPosition().x + getSize().width / 2;
118     }
119 
120     protected int getCenterY() {
121         return getPosition().y + getSize().height / 2;
122     }

124 }
