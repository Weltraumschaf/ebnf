11  package de.weltraumschaf.ebnf.gfx.shapes;
12  
13  import de.weltraumschaf.ebnf.gfx.Point;
14  import de.weltraumschaf.ebnf.gfx.StringPainter;
15  import de.weltraumschaf.ebnf.gfx.Strokes;
16  import java.awt.Color;
17  import java.awt.Dimension;
18  import java.awt.Graphics2D;
19  
20  /**
21   *
22   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
23   */
24  public class Identifier extends AbstractTextShape implements Shape {
25  
26      public Identifier(final String text) {
27          super(text, StringPainter.SANSERIFIT);
28      }
29  
30      private Dimension boxSize = null;
31  
32      protected Dimension calcBoxSize(final Graphics2D graphic) {
33          final Dimension textSize = calculateTextSize(graphic);
34          return new Dimension(textSize.width + H_PADDING * 2, textSize.height);
35      }
36  
37      @Override
38      public void adjust(final Graphics2D graphic) {
39          boxSize = calcBoxSize(graphic);
40          setSize(new Dimension(calculateWidth(boxSize.width), DEFAULT_HEIGHT));
41      }
42  
43      @Override
44      public void paint(final Graphics2D graphic) {
45          if (null == boxSize) {
46              adjust(graphic);
47          }
48  
49          final Point pos     = getPosition();
50          final Dimension size = getSize();
51          final int vCenter   = getCenterY();
52          final int hPaddingp = (size.width - boxSize.width) / 2;
53          final int vPadding  = (size.height - boxSize.height) / 2;
54          final Point rectanglePosition = new Point(pos.x + hPaddingp, pos.y + vPadding),
55                      inLineStart       = new Point(pos.x, vCenter),
56                      inLineEnd         = new Point(pos.x + hPaddingp, vCenter),
57                      outLineStart      = new Point(pos.x + hPaddingp + boxSize.width, vCenter),
58                      outLineEnd        = new Point(pos.x + size.width, vCenter);
59  
60          super.paint(graphic);
61          backupColorAndStroke(graphic);
62          graphic.setColor(Color.BLACK);
63          graphic.setStroke(Strokes.createForBox());
64          graphic.drawRect(rectanglePosition.x, rectanglePosition.y, boxSize.width, boxSize.height);
65          graphic.setStroke(Strokes.createForRail());
66          graphic.drawLine(inLineStart.x, inLineStart.y, inLineEnd.x, inLineEnd.y);
67          graphic.drawLine(outLineStart.x, outLineStart.y, outLineEnd.x, outLineEnd.y);
68          final StringPainter textPainter = createStringPainter(graphic);
69          textPainter.drawCenteredString(getText(), rectanglePosition, boxSize);
70          resotreColorAndStroke(graphic);
71      }
72  
73  }
