12  package de.weltraumschaf.ebnf.gfx.shapes;
13  
14  import com.google.common.collect.Lists;
15  import de.weltraumschaf.ebnf.gfx.Point;
16  import java.awt.Dimension;
17  import java.awt.Graphics2D;
18  import java.util.List;
19  
20  /**
21   *
22   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
23   */
24  public class GridLayout extends AbstractLayout {
25  
26      private final List<ColumnLayout> columns = Lists.newArrayList();
27  
28      public int countCols() {
29          return columns.size();
30      }
31  
32      public int counRows() {
33          return countCols() == 0
34                 ? 0
35                 : columns.get(0).countShapes();
36      }
37  
38      public ColumnLayout get(final int columnIndex) {
39          ColumnLayout col;
40  
41          try {
42              col = columns.get(columnIndex);
43          } catch (IndexOutOfBoundsException ex) {
44              col = null;
45          }
46  
47          if (null == col) {
48              // @todo test exception
49              throw new IllegalArgumentException(
50                      String.format("The column at columnIndex %d is not present!", columnIndex));
51          }
52  
53          return col;
54      }
55  
56      public Shape get(final int columnIndex, final int rowIndex) {
57          Shape shape;
58  
59          try {
60              shape = get(columnIndex).get(rowIndex);
61          } catch (IndexOutOfBoundsException ex) {
62              shape = null;
63          }
64  
65          if (null == shape) {
66              // @todo test exception
67              throw new IllegalArgumentException(
68                      String.format("The shape at columnIndex %d and rowIndex %d is not present!",
69                                    columnIndex, rowIndex));
70          }
71  
72          return shape;
73      }
74  
75      public GridLayout set(final int columnIndex, final int rowIndex, final Shape shape) {
76          final int colCount = countCols();
77  
78          if (columnIndex >= colCount) {
79              for (int i = colCount; i < columnIndex + 1; ++i) {
80                  columns.add(ColumnLayout.newColumnLayout());
81              }
82          }
83  
84          final ColumnLayout col = columns.get(columnIndex);
85          col.set(rowIndex, shape);
86          return this;
87      }
88  
89      public GridLayout append(final ColumnLayout... cols) {
90          for (ColumnLayout cl : cols) {
91              append(cl);
92          }
93  
94          return this;
95      }
96  
97      public GridLayout append(final ColumnLayout col) {
98          columns.add(col);
99          return this;
100     }
101 
102     @Override
103     public void paint(final Graphics2D graphic) {
104         final Point pos = getPosition();
105         int currentX = pos.x;
106 
107         for (ColumnLayout col : columns) {
108             col.setPosition(pos.setX(currentX));
109             col.setDebug(isDebug());
110             col.paint(graphic);
111             currentX += col.getSize().width;
112         }
113     }
114 
115     @Override
116     public void adjust(final Graphics2D graphic) {
117         int width = 0;
118         int height = 0;
119 
120         for (ColumnLayout col : columns) {
121             col.adjust(graphic);
122             final  Dimension colSize = col.getSize();
123             width  += colSize.width;
124             height = Math.max(height, colSize.height);
125         }
126 
127         setSize(new Dimension(width, height));
128     }
129 
130 }
