12  package de.weltraumschaf.ebnf.gfx;
13  
14  import java.awt.Font;
15  
16  /**
17   * Collection of used fonts with default size.
18   *
19   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
20   */
21  public enum Fonts {
22      /**
23       * Used for terminals.
24       */
25      MONOSPACED("Monospaced", Font.PLAIN),
26      /**
27       * Used for rule names.
28       */
29      SANSERIF("Sanserif", Font.PLAIN),
30      /**
31       * Used for identifiers.
32       */
33      SANSERIFIT("Sanserif", Font.ITALIC);
34  
35      /**
36       * Font name.
37       */
38      private final String name;
39      /**
40       * Font style.
41       */
42      private final int style;
43      /**
44       * Font size.
45       */
46      private int size;
47      /**
48       * Caches font object for particular font size.
49       */
50      private Font font = null;
51  
52      /**
53       * Default font size.
54       */
55      public static final int DEFAULT_SIZE = 6;
56  
57      /**
58       * Font with default size.
59       *
60       * @param name  Font name.
61       * @param style Font style.
62       */
63      Fonts(final String name, final int style) {
64          this(name, style, DEFAULT_SIZE);
65      }
66  
67      /**
68       * Font with custom size.
69       *
70       * @param name  Font name.
71       * @param style Font style.
72       * @param size  Font size.
73       */
74      Fonts(final String name, final int style, final int size) {
75          this.name  = name;
76          this.style = style;
77          this.size  = size;
78      }
79  
80      /**
81       * Creates an AWT font object.
82       *
83       * Returns always same object until a different size is creates.
84       *
85       * @return
86       */
87      public Font create() {
88          if (null == font) {
89              font = new Font(name, style, size);
90          }
91          return font;
92      }
93  
94      /**
95       * Creates a font object with a new size.
96       *
97       * @param size New font size.
98       * @return
99       */
100     public Font create(final int size) {
101         this.size = size;
102         font = null;
103         return create();
104     }
105 
106 }
