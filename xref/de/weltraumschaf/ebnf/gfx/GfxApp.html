30  public class GfxApp {

32      private static final int WIDTH = 800;
33      private static final int HEIGHT = 600;
34  
35      boolean debug = false;
36  
37      private final String[] args;
38      private final JFrame frame = new JFrame();
39  
40      public GfxApp(final String[] args) {
41          this.args = args.clone();
42      }
43  
44      public int execute() {
45          String type = "";
46  
47          for (String arg : args) {
48              if (arg.equalsIgnoreCase("swing") || arg.equalsIgnoreCase("img")) {
49                  type = arg.toLowerCase(); //NOPMD
50              } else if (arg.equalsIgnoreCase("-d") || arg.equalsIgnoreCase("--debug")) {
51                  debug = true;
52              }
53          }
54  
55          if (type.equalsIgnoreCase("swing")) {
56              createAndShowGui();
57          } else if (type.equalsIgnoreCase("img")) {
58              createAndSaveImage();
59          } else {
60              System.out.println(String.format("Give either 'swing' or 'img'! Given was '%s'.",
61                                               type));
62              return -1;
63          }
64  
65          return 0;
66      }
67  
68      private void createAndShowGui() {
69          //Schedule a job for the event-dispatching thread:
70          //creating and showing this application's GUI.
71          SwingUtilities.invokeLater(new Runnable() {
72              @Override
73              public void run() {
74                  frame.setTitle("Railroad");
75                  frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
76                  frame.setSize(new Dimension(WIDTH, HEIGHT));
77                  frame.setLocationRelativeTo(null);
78                  frame.setVisible(true);
79                  final RailroadDiagramPanel panel = new RailroadDiagramPanel(
80                          createDiagram(frame.getGraphics()));
81                  frame.add(panel);
82                  frame.validate();
83              }
84          });
85      }
86  
87      private void createAndSaveImage() {
88          final RailroadDiagramImage img = new RailroadDiagramImage(WIDTH, HEIGHT,
89                                                                    new File("./test.png"));
90          img.setDiagram(createDiagram(img.getGraphics()));
91          img.paint();
92          try {
93              img.save();
94          } catch (IOException ex) {
95              System.out.println("Can't write file!");
96          }
97      }

99      private GridLayout createValueDiagram(final Graphics2D graphics) {
100         final GridLayout value = grid();
101         value.append(
102             column().append(
103                 rule("value"),
104                 start(),
105                 empty(),
106                 empty(),
107                 empty(),
108                 empty()
109             )
110         );
111         value.set(1, 1, choice(
112             identifier("string"),
113             identifier("number"),
114             identifier("object"),
115             terminal("true"),
116             terminal("false")
117         ));
118         value.append(
119             column().append(
120                 empty(),
121                 end(),
122                 empty(),
123                 empty(),
124                 empty(),
125                 empty()
126             )
127         );
128         value.adjust(graphics);
129         return value;
130     }

132     private GridLayout createObjectDiagram(final Graphics2D graphics) {
133         final GridLayout object = grid();
134         final Sequence innerSequence = sequence(identifier("string"),
135                                                 terminal(":"),
136                                                 identifier("value"));
137         innerSequence.adjust(graphics);
138         final Loop loop = loop(innerSequence, terminal(","));
139         loop.adjust(graphics);
140         final Option option = option(loop);
141 
142         object.append(
143             column().append(
144                 rule("object"),
145                 start(),
146                 empty(),
147                 empty()
148             )
149         );
150         object.append(
151             column().append(
152                 empty(),
153                 terminal("{"),
154                 empty(),
155                 empty()
156             )
157         );
158         object.append(
159             column().append(
160                 empty(),
161                 option
162             )
163         );
164         object.append(
165             column().append(
166                 empty(),
167                 terminal("}"),
168                 empty(),
169                 empty()
170             )
171         );
172         object.append(
173             column().append(
174                 empty(),
175                 end(),
176                 empty(),
177                 empty()
178             )
179         );
180         object.adjust(graphics);
181         return object;
182     }

184     private RailroadDiagram createDiagram(final Graphics graphics) {
185         return createDiagram((Graphics2D) graphics);
186     }
187 
188     private RailroadDiagram createDiagram(final Graphics2D graphics) {
189         final Point offset = new Point(20, 20);
190         final RailroadDiagram diagram = new RailroadDiagram();
191         diagram.setDebug(debug);
192         final GridLayout value = createValueDiagram(graphics);
193         value.setPosition(offset);
194         diagram.add(value);
195 
196         final GridLayout object = createObjectDiagram(graphics);
197         object.setPosition(new Point(offset.x, offset.y + value.getSize().height));
198         diagram.add(object);
199 
200         return diagram;
201     }
