12  package de.weltraumschaf.ebnf.gfx;
13  
14  import com.google.common.base.Objects;
15  
16  /**
17   * Represents a geometric line with a {@link Point start} and {@link Point end}.
18   *
19   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
20   */
21  public class Line {
22  
23      /**
24       * Start point of line.
25       */
26      public final Point start;
27      /**
28       * End point of line.
29       */
30      public final Point end;
31  
32      /**
33       * Initializes a line with both start and end with points to (0, 0).
34       */
35      public Line() {
36          this(new Point(), new Point());
37      }
38  
39      /**
40       * Designated constructor.
41       *
42       * @param start
43       * @param end
44       */
45      public Line(final Point start, final Point end) {
46          this.start = start;
47          this.end = end;
48      }
49  
50      /**
51       * String representation.
52       *
53       * @return
54       */
55      @Override
56      public String toString() {
57          return Objects.toStringHelper(this).add("start", start).add("end", end).toString();
58      }
59  
60  }
