12  package de.weltraumschaf.ebnf.ast;
13  
14  import de.weltraumschaf.ebnf.ast.nodes.*;
15  
16  /**
17   *
18   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
19   */
20  public final class NodeFactory {
21  
22      private NodeFactory() { }
23  
24      public static Node newNode(final NodeType type, final Node parent) {
25          switch (type) {
26              case CHOICE:
27                  return Choice.newInstance(parent);
28              case COMMENT:
29                  return Comment.newInstance(parent);
30              case IDENTIFIER:
31                  return Identifier.newInstance(parent);
32              case LOOP:
33                  return Loop.newInstance(parent);
34              case OPTION:
35                  return Option.newInstance(parent);
36              case RULE:
37                  return Rule.newInstance(parent);
38              case SEQUENCE:
39                  return Sequence.newInstance(parent);
40              case TERMINAL:
41                  return Terminal.newInstance(parent);
42              default:
43                  // This may happen if someone adds new NodeType enum w/o adding here.
44                  throw new IllegalArgumentException(String.format("Unsupported node type '%s'!", type));
45          }
46      }
47  
48      public static Node newNode(final NodeType type) {
49          if (NodeType.SYNTAX == type) {
50              return Syntax.newInstance();
51          } else if (NodeType.NULL == type) {
52              return Null.getInstance();
53          } else {
54              return newNode(type, Null.getInstance());
55          }
56      }
57  }
