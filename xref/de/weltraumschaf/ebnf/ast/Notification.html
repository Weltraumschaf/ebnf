6   /**
7    * Used to notify collected AST errors.

9    * @author Sven Strittmatter <weltraumschaf@googlemail.com>

11  public class Notification {
12  
13      /**
14       * Collected arrays.
15       *
16       * @var
17       */
18      private final List<String> errors = new ArrayList<String>();
19  
20      /**
21       * Collect an line of error.
22       *
23       * The first parameter is a sprintf style format string.
24       *
25       * Example:
26       * error(format, arg1, arg2 .. argN)
27       *
28       * @param format Format string.
29       * @param args   Variable number of objects referenced in the format string.
30       */
31      public void error(final String format, final Object... args) {
32          errors.add(String.format(format, args));
33      }
34  
35      /**
36       * Returns true if no error was collected.
37       *
38       * @return
39       */
40      public boolean isOk() {
41          return errors.isEmpty();
42      }
43  
44      /**
45       * Returns all errors concatenated as string.
46       *
47       * @return string
48       */
49      public String report() {
50          if (isOk()) {
51              return "";
52          }
53  
54          final StringBuilder report = new StringBuilder();
55  
56          for (int i = 0; i < errors.size(); ++i) {
57              if (i > 0) {
58                  report.append('\n');
59              }
60  
61              report.append(errors.get(i));
62          }
63          return report.toString();
64      }
