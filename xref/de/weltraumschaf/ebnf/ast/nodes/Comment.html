77          try {
78              final Comment terminal = (Comment) other;
79  
80              if (!getAttribute(ATTR_VALUE).equals(terminal.getAttribute(ATTR_VALUE))) {
81                  result.error("Comment value mismatch: '%s' != '%s'!", getAttribute(ATTR_VALUE),
82                                                                        terminal.getAttribute(ATTR_VALUE));
83              }
84          } catch (ClassCastException ex) {
85              result.error(

87                  getClass(),
88                  other.getClass()
89              );
90          }
91      }
