69      @Override

71          super.probeEquivalence(other, result);
72  
73          final Rule rule = (Rule) other;
74  
75          if (!getAttribute(ATTRIBUTE_NAME).equals(rule.getAttribute(ATTRIBUTE_NAME))) {
76              result.error("Names of rule differs: '%s' != '%s'!", getAttribute(ATTRIBUTE_NAME),
77                                                                   rule.getAttribute(ATTRIBUTE_NAME));

79      }
