116     private void runCliApp(final CliOptions options) {
117         if (!options.hasSyntaxFile()) {
118             println("No syntax file given!");
119             exit(ExitCode.NO_SYNTAX);
120         }
121 
122         final String fileName = options.getSyntaxFile();
123 
124         try {
125             final Parser parser  = Factory.newParserFromSource(new File(fileName), fileName);
126             final Syntax ast     = parser.parse();
127 
128             if (options.isTextTree()) {
129                 final TextSyntaxTree visitor = new TextSyntaxTree();
130                 ast.accept(visitor);
131                 println(visitor.getText());
132             }
133         } catch (SyntaxException ex) {
134             println("Syntax error: " + ex.getMessage());
135 
136             if (options.isDebug()) {
137                 ex.printStackTrace(System.out);
138             }
139 
140             exit(ExitCode.SYNTAX_ERROR);
141         } catch (FileNotFoundException ex) {
142             println(String.format("Can not read syntax file '%s'!", fileName));
143 
144             if (options.isDebug()) {
145                 ex.printStackTrace(System.out);
146             }
147 
148             exit(ExitCode.READ_ERROR);
149         } catch (IOException ex) {
150             println(String.format("Can not read syntax file '%s'!", fileName));
151 
152             if (options.isDebug()) {
153                 ex.printStackTrace(System.out);
154             }
155 
156             exit(ExitCode.READ_ERROR);
157         }
158 
159         exit(ExitCode.OK);
160     }
