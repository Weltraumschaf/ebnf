12      private final class Parser {
13          private static final String OPT_SYNTAX    = "s";
14          private static final String OPT_OUTPUT    = "o";
15          private static final String OPT_FORMAT    = "f";
16          private static final String OPT_TEXT_TREE = "t";
17          private static final String OPT_DEBUG     = "d";
18          private static final String OPT_HELP      = "h";

20          public void parse(final String[] args) throws ParseException {
21              final CommandLineParser parser = new PosixParser();
22              final CommandLine cmd = parser.parse(options, args);
23  
24              optSyntax(cmd);
25              optFormat(cmd);
26              optOutputFile(cmd);
27              optTextTree(cmd);
28              optDebug(cmd);
29              optHelp(cmd);
30          }
31  
32          private void optSyntax(final CommandLine cmd) {
33              if (cmd.hasOption(OPT_SYNTAX)) {
34                  syntaxFile = cmd.getOptionValue(OPT_SYNTAX);
35              }
36          }
37  
38          private void optFormat(final CommandLine cmd) {
39              if (cmd.hasOption(OPT_FORMAT)) {
40                  final String formatOption = cmd.getOptionValue(OPT_FORMAT);
41  
42                  if (formatOption.equalsIgnoreCase("xml")) {
43                      outputFormat = OutputFormat.XML;
44                  } else if (formatOption.equalsIgnoreCase("jpg")) {
45                      outputFormat = OutputFormat.JPG;
46                  } else if (formatOption.equalsIgnoreCase("gif")) {
47                      outputFormat = OutputFormat.GIF;
48                  }
49              }
50          }
51  
52          private void optOutputFile(final CommandLine cmd) {
53              if (cmd.hasOption(OPT_OUTPUT)) {
54                  outputFile = cmd.getOptionValue(OPT_OUTPUT);
55              } else {
56                  outputFile = syntaxFile.replace(".ebnf", "." + outputFormat.name().toLowerCase());
57              }
58          }
59  
60          private void optTextTree(final CommandLine cmd) {
61              if (cmd.hasOption(OPT_TEXT_TREE)) {
62                  textTree = true;
63              }
64          }
65  
66          private void optDebug(final CommandLine cmd) {
67              if (cmd.hasOption(OPT_DEBUG)) {
68                  debug = true;
69              }
70          }
71  
72          private void optHelp(final CommandLine cmd) {
73              if (cmd.hasOption(OPT_HELP)) {
74                  help = true;
75              }
76          }
77      }

79      private final Options options;
80  
81      private OutputFormat outputFormat = OutputFormat.JPG;
82      private String syntaxFile   = "";
83      private String outputFile   = "";
84      private boolean textTree = false;
85      private boolean debug    = false;
86      private boolean help     = false;

88      public CliOptions() {
89          options = new Options();

91          options.addOption(Parser.OPT_SYNTAX, true, "EBNF syntax file to parse.");    // required
92          options.addOption(Parser.OPT_OUTPUT, true, "Output file name.");
93          options.addOption(Parser.OPT_FORMAT, true, "Output format: xml, jpg, gif.");
94          // w/o argument
95          options.addOption(Parser.OPT_TEXT_TREE,
96                            false,
97                            "Prints textual representation of the syntax tree to stdout.");
98          options.addOption(Parser.OPT_DEBUG,     false, "Enables debug output.");
99          options.addOption(Parser.OPT_HELP,      false, "This help.");
100     }
