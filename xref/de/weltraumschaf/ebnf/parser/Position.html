4    * Represents a token position in the source string.

11  public class Position {

13      /**
14       * File of the source string.
15       */
16      private String file;
17      /**
18       * Line of occurrence.
19       */
20      private int line;
21      /**
22       * Column of occurrence.
23       */
24      private int column;
25  
26      /**
27       * Initializes without file.
28       *
29       * @param line Line of occurrence.
30       * @param column Column of occurrence.
31       */
32      public Position(final int line, final int column) {
33          this(line, column, null);
34      }
35  
36      /**
37       * Dedicated constructor initializes immutable object.
38       *
39       * File is optional e.g. if string is parsed directly without any file.
40       *
41       * @param line Line of occurrence.
42       * @param column Column of occurrence.
43       * @param file Optional file name.
44       */
45      public Position(final int line, final int column, final String file) {
46          this.line   = line;
47          this.column = column;
48          this.file   = file;
49      }
50  
51      /**
52       * Returns the file name of the source.
53       *
54       * May be null.
55       *
56       * @return Filename as string.
57       */
58      public String getFile() {
59          return file;
60      }
61  
62      /**
63       * Returns line of occurrence in source.
64       *
65       * @return Line position.
66       */
67      public int getLine() {
68          return line;
69      }
70  
71      /**
72       * Returns column of occurrence in source.
73       *
74       * @return Column position.
75       */
76      public int getColumn() {
77          return column;
78      }
79  
80      /**
81       * Returns human readable string representation.
82       *
83       * @return String representation.
84       */
85      @Override
86      public String toString() {
87          final StringBuilder str = new StringBuilder();
88  
89          if (getFile() != null) {
