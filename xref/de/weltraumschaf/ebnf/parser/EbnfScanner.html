10  /**
11   * Scans an input string for EBNF syntax tokens.
12   *
13   * This class implements a standard lexical scanner pattern with one
14   * character lookahead and iterator interface for receiving token by token.
15   * On lexical syntax errors a SyntaxException will be thrown.
16   *
17   * Example:
18   * <code>
19   * BufferedReader grammar = new BufferedReader(new StringReader("...")); // the EBNF grammar
20   *
21   * try {
22   *     Scanner scanner = new Scanner(grammar);
23   *
24   *     while (scanner->hasNextToken()) {
25   *         scanner->nextToken();
26   *         Token token = scanner->currentToken();
27   *         doSomething(token);
28   *     }
29   * } catch (SyntaxtError ex) {
30   *      logError(ex);
31   * }
32   * </code>
33   *
34   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
35   */
36  public class EbnfScanner implements Scanner {

137     /**
138      * Increments the character cursor.
139      *
140      * @throws IOException On IO errors caused by the input reader.
141      */
142     private void nextCharacter() throws IOException {
143         if (currentCharacter > -1 && EOF == getCurrentCharacter()) {
144             return;
145         }
146 
147         if (currentCharacter + 1 >= buffer.length()) {
148             final int chr = input.read();
149 
150             if (-1 == chr) {
151                 atEof = true;
152             } else {
153                 buffer.append((char) chr);
154             }
155         }
156 
157         currentCharacter++;
158         column++;
159     }

161     /**
162      * Returns the character at the current cursor from the input stream.
163      *
164      * @return The current character.
165      */
166     private char getCurrentCharacter() {
167         try {
168             return buffer.charAt(currentCharacter);
169         } catch (StringIndexOutOfBoundsException ex) {
170             if (atEof) {
171                 return EOF;
172             } else {
173                 throw new IllegalStateException("Invoke Scanner#nextCharacter() first!", ex);
174             }
175         }
176     }

188     /**
189      * Returns next character without advancing the cursor.
190      *
191      * @return
192      */
193     private char peekCharacter() throws IOException {
194         char character = EOF;
195 
196         if (hasNextCharacter()) {
197             nextCharacter();
198             character = getCurrentCharacter();
199             backupCharacter();
200         }
201 
202         return character;
203     }

289     /**
290      * Returns the next token without advancing the internal pointer (aka. lookahead).
291      *
292      * A call to {@link Scanner#nextToken()} will return this token ahead.
293      *
294      * @return
295      * @throws SyntaxException On syntax errors.
296      * @throws IOException     On input stream IO errors.
297      */
298     @Override
299     public Token peekToken() throws SyntaxException, IOException {
300         nextToken();
301         final Token token = getCurrentToken();
302         currentToken--;
303 
304         // @todo May be this is not necessary.
305         if (currentToken < 0) {
306             currentToken = 0;
307         }
308 
309         return token;
310     }

312     /**
313      * Start the scanning of the next token.
314      *
315      * This method should be called until {hasNextToken()} returns false.
316      *
317      * @throws SyntaxException On syntax errors.
318      * @throws IOException     On input stream IO errors.
319      */
320     @Override
321     public void nextToken() throws SyntaxException, IOException {
322         if (currentToken > -1 && currentToken < (tokens.size() - 1)) {
323             // recover backtracked tokens.
324             currentToken++;
325             return;
326         }
327 
328         while (hasNextCharacter()) {
329             nextCharacter();
330 
331             if (isAlpha(getCurrentCharacter())) {
332                 tokens.add(scanIdentifier());
333                 currentToken++;
334                 return;
335             } else if (isQuote(getCurrentCharacter())) {
336                 tokens.add(scanLiteral());
337                 currentToken++;
338                 return;
339             } else if (isOperator(getCurrentCharacter())) {
340                 if ('(' == getCurrentCharacter() && '*' == peekCharacter()) {
341                     tokens.add(scanComment());
342                 } else {
343                     tokens.add(scanOperator());
344                 }
345 
346                 currentToken++;
347                 return;
348             } else if (isWhiteSpace(getCurrentCharacter())) { // NOPMD
349                 // Ignore white spaces.
350             } else {
351                 raiseError(String.format("Invalid character '%s' as %s",
352                                          getCurrentCharacter(),
353                                          createPosition()));
354             }
355 
356             checkNewline();
357         }
358 
359         tokens.add(new Token(TokenType.EOF, file, createPosition()));
360         currentToken++;
361     }

364      * Checks if the current character is a new line character (\n or \r)
365      * and if it is increments the line counter and resets the column counter to 0.
366      *
367      * @return

369     private void checkNewline() {
370         if ('\n' == getCurrentCharacter() || '\r' == getCurrentCharacter()) {
371             line++;
372             column = 0;
373         }
374     }

376     /**
377      * Scans an identifier [a-zA-Z\-_].
378      *
379      * @return
380      */
381     private Token scanIdentifier() throws IOException {
382         final Position position = createPosition();
383         final StringBuilder value = new StringBuilder();
384         value.append(getCurrentCharacter());
385 
386         while (hasNextCharacter()) {
387             nextCharacter();
388 
389             if (isAlphaNum(getCurrentCharacter())
390                     || isEquals(getCurrentCharacter(), SPECIAL_CHARS)) {
391                 value.append(getCurrentCharacter());
392             } else {
393                 backupCharacter();
394                 break;
395             }
396         }
397 
398         return new Token(TokenType.IDENTIFIER, value.toString(), position);
399     }

401     /**
402      * Scans a literal (any character inside single or double quotes.
403      *
404      * @return
405      */
406     private Token scanLiteral() throws IOException {
407         final Position position = createPosition();
408         final char start = getCurrentCharacter();
409         final StringBuilder value = new StringBuilder();
410         value.append(start);
411 
412         while (hasNextCharacter()) {
413             nextCharacter();
414             value.append(getCurrentCharacter());
415 
416             // Ensure that a lieral opened with " is not temrinated by ' and vice versa.
417             if (isQuote(getCurrentCharacter()) && getCurrentCharacter() == start) {
418                 break;
419             }
420         }
421 
422         return new Token(TokenType.LITERAL, value.toString(), position);
423     }

425     /**
426      * Scans a comment (any character inside '(*' and '*)'.
427      *
428      * @return
429      */
430     private Token scanComment() throws IOException {
431         final Position postition = createPosition();
432         final StringBuilder value = new StringBuilder();
433         value.append(getCurrentCharacter());
434 
435         while (hasNextCharacter()) {
436             nextCharacter();
437             value.append(getCurrentCharacter());
438 
439             if ('*' == getCurrentCharacter() && ')' == peekCharacter()) {
440                 nextCharacter();
441                 value.append(getCurrentCharacter());
442                 break;
443             }
444 
445             checkNewline(); // // Comments cann be multiline.
446         }
447 
448         return new Token(TokenType.COMMENT, value.toString(), postition);
449     }

451     /**
452      * Scans an operator.
453      *
454      * @return
455      */
456     private Token scanOperator() throws SyntaxException, IOException {
457         final Position position   = createPosition();
458         final StringBuilder value = new StringBuilder();
459         value.append(getCurrentCharacter());
460 
461         final char peek = peekCharacter();
462         TokenType type = null;
463 
464         switch (getCurrentCharacter()) {
465             case ':': {
466                 if ('=' == peek) {
467                     nextCharacter();
468                     value.append(getCurrentCharacter());
469                     nextCharacter();
470 
471                     if (getCurrentCharacter() != '=') {
472                         raiseError(String.format("Expecting '=' but seen '%s'",
473                                    getCurrentCharacter()));
474                     }
475 
476                     value.append(getCurrentCharacter());
477                     type = TokenType.ASIGN;
478                 } else {
479                     type = TokenType.ASIGN;
480                 }
481 
482                 break;
483             }
484             case '=': type = TokenType.ASIGN; break;
485             case '.': { // range or end of rule
486                 if ('.' == peek) {
487                     nextCharacter();
488                     value.append(getCurrentCharacter());
489                     type = TokenType.RANGE;
490                 } else {
491                     type = TokenType.END_OF_RULE;
492                 }
493 
494                 break;
495             }
496             case ';': type = TokenType.END_OF_RULE; break;
497             case '(': type = TokenType.L_PAREN; break;
498             case '[': type = TokenType.L_BRACK; break;
499             case '{': type = TokenType.L_BRACE; break;
500             case ')': type = TokenType.R_PAREN; break;
501             case ']': type = TokenType.R_BRACK; break;
502             case '}': type = TokenType.R_BRACE; break;
503             case '|': type = TokenType.CHOICE; break;
504             default: {
505                 raiseError(String.format("Unexpected operator character '%s'",
506                                          getCurrentCharacter()));
507             }
508         }
509 
510         return new Token(type, value.toString(), position);
511     }
